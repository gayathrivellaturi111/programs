	<button onclick="myFunction()">

	</button>
	
	<script type="text/javascript">
		function myFunction() {
			var link = document.querySelector("a");
			link.getAttribute("href");
			link.setAttribute("href",
				"https://www.geeksforgeeks.org");
			link.textContent = "Go to GeeksforGeeks";
		}
	</script>
